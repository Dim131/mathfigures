 \LT{step2}{The altitudes $\rm BE$ and $\rm \Gamma Z$ intersect at $\rm H$.}

 \begin{itemize}
   \LT{step4}{\item The quadrilateral $\rm BZE\Gamma$ is \href{http://en.wikipedia.org/wiki/Cyclic quadrilateral}{cyclic}, because $\angle {\rm BZ\Gamma} = \angle {\rm \Gamma EB} = 90^\circ$.}

   \LT{step6}{\item The quadrilateral $\rm AZHE$ is cyclic because opposite angles $\angle{\rm AEH}$ and $\angle{\rm AZH}$ are \href{https://en.wikipedia.org/wiki/Supplementary\_angles}{supplementary} (since $\angle{\rm AEH} + \angle{\rm AZH} = 90^\circ + 90^\circ = 180^\circ$).}

   \LT{step8}{\item The angles $\angle{\rm AHZ}$ and $\angle{\rm \Delta H\Gamma}$ are equal as \href{https://en.wikipedia.org/wiki/Angle\#Vertical\_and\_adjacent\_angle\_pairs}{vertical angles}, i.e. $\angle{\rm AHZ} = \angle{\rm \Delta H\Gamma} = \hat{\varphi}$.}

  \end{itemize}
\LT{step10}{We conclude that the three altitudes pass through the same point.}
