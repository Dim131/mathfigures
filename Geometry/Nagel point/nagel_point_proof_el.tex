\begin{document}

\begin{tikzpicture}[scale=.75]

\tkzDefTriangleCenter[nagel](A,B,C)

\tkzDefPointBy[projection=onto A--C](Ja)\tkzGetPoint{JJB}


\tkzFillPolygon[fill=ShapeClr,fill opacity=0.1](A,B,C)
\tkzDrawPoints(Ja,Ta)
\tkzDrawLines[add=0 and 1.75,dashed](A,B)
\tkzDrawLines[add=1.75 and 0,dashed](C,A)
\tkzDrawPoints[size=0](AA)

\tkzClipBB
\tkzDrawCircles[line width=0.5pt,color=ParacircleClr,fill=ParacircleClr,fill opacity=0.1](Ja,Ta)

% Double draw these.
\tkzDrawLines[add=0 and 1.75,dashed](A,B)
\tkzDrawLines[add=1.75 and 0,dashed](C,A)


\tkzDrawSegments[line width=0.5pt,color=black,dashed,dash pattern=on 1pt off 1.75pt](Ja,Ta Ja,JJB Ja,JJC)

\tkzMarkRightAngles[line width=0.5pt, size=.15,color=AngleClr,fill=AngleClr,fill opacity=0.1](Ja,Ta,C Ja,JJB,A Ja,JJC,A)

\tkzDrawPolygon[color=ShapeClr](A,B,C)

\tkzDrawPoints[size=3](Ja)
\tkzDrawPoints[size=2](Ta,JJB,JJC)
\tkzDrawPoints[size=3](B,C,A)

\tkzLabelPoint[above](A){$\rm A$}
\tkzLabelPoint[left](B){$\rm B$}
\tkzLabelPoint[right](C){$\rm \Gamma$}

\tkzLabelPoint[above](Ta){$\rm I_A'$}

\tkzLabelPoint[above left](JJC){$\rm I_A''$}
\tkzLabelPoint[above right](JJB){$\rm I_A'''$}

\tkzLabelPoint[below](Ja){$\rm J_A$}
\tkzLabelPoint[above right](Jb){$\rm J_B$}

\tkzLabelSegments[above](B,Ta){$x$}

\tkzLabelSegments[left](B,JJC){$x$}
\tkzLabelSegments[right](C,JJB){$y$}

\tkzLabelSegments[left](A,B){$\gamma$}
